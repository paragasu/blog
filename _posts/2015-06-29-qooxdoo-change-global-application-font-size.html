setFontSize: function(size)
{
var doc = this.getRoot();
var manager = qx.theme.manager.Font.getInstance();
var defaultFont = manager.resolve(doc.getFont());
var newFont = defaultFont.clone().setSize(parseInt(size));

doc.setFont(newFont);
},
